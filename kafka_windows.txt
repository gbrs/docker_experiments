Запустить Docker Desktop

cd J:\docker
mkdir kafka
cd .\kafka\

Создать здесь файл docker-compose.yml:
```yaml
services:
  zookeeper:
    image: confluentinc/cp-zookeeper:latest
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181

  kafka:
    image: confluentinc/cp-kafka:latest
    depends_on:
      - zookeeper
    ports:
      - "9092:9092"
    environment:
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://localhost:9092
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1

  # GUI
  akhq:
    image: tchiotludo/akhq
    depends_on:
      - kafka
    ports:
      - "8080:8080" # Прокидываем веб-интерфейс на 8080 порт Windows
    environment:
      AKHQ_CONFIGURATION: |
        akhq:
          connections:
            docker-kafka-server:
              properties:
                bootstrap.servers: "kafka:9092" # Обратите внимание: внутри Docker-сети используется имя сервиса 'kafka'
```

docker-compose up -d  # запустит ZooKeeper, Kafka и AKHQ в фоновом режиме

На http://localhost:8080 будет интерфейс AKHQ
Входим и создаем (кнопка справа-внизу) новый топик: test-topic
